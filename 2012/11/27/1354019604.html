とりあえず、search_users()で自分を検索してみる。

~~~~ {.syntax-highlight} #!/usr/bin/env python

import sys import tweepy import inspect

CONSUMER_KEY = '**' CONSUMER_SECRET = '' ACCESS_KEY = '' ACCESS_SECRET = '**'

auth = tweepy.OAuthHandler(CONSUMER_KEY, CONSUMER_SECRET) auth.set_access_token(ACCESS_KEY, ACCESS_SECRET) api = tweepy.API(auth) users = api.search_users(sys.argv[1])

for p in inspect.getmembers(users[0]): print p ~~~~
